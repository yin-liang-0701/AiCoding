class ListMemosUseCase:
    def __init__(self, memo_repository):
        self.memo_repository = memo_repository

    def execute(self, user_id, tag=None):  # 增加tag参数
        if tag:
            # 按标签过滤（需确保repository支持用户隔离）
            return self.memo_repository.find_by_tag(user_id, tag)
        else:
            return self.memo_repository.find_all(user_id)
```

#### (3) repository.py - 增强查询能力

# 在SqliteMemoRepository类中确保有：
def find_by_tag(self, user_id, tag_name):
    """按用户ID和标签名过滤备忘录"""
    conn = self._get_conn()
    cursor = conn.cursor()
    cursor.execute("""
        SELECT m.* FROM memos m
        JOIN memo_tags mt ON m.id = mt.memo_id
        JOIN tags t ON mt.tag_id = t.id
        WHERE m.user_id = ? AND t.name = ?
    """, (user_id, tag_name))
    rows = cursor.fetchall()
    return [self._row_to_memo(row) for row in rows]
```
