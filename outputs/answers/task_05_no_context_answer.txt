```go
func ListResources(tagFilter string) ([]Resource, error) {
    resources, err := store.GetAll()
    if err != nil {
        return nil, err
    }
    
    if tagFilter != "" {
        filtered := make([]Resource, 0)
        for _, r := range resources {
            // 假设 Resource 有 Tags 字段 []string
            for _, tag := range r.Tags {
                if tag == tagFilter {
                    filtered = append(filtered, r)
                    break
                }
            }
        }
        return filtered, nil
    }
    
    return resources, nil
}
```
